\frame{{Recap: deterministic growth}

\bi
\item[]<1->
\be
dx=f(x) d(\gamma t)
\ee
\item[]<2->
\hspace{-1.7cm}
\be
\text{$\to$ linear growth of} \int_{x(t)}^{x(t+\Dt)} \frac{1}{f(x)}dx=\gamma \Dt.
\ee
\item[]<3->

\be \text{Re-write as } v(x(t+\Dt))-v(x(t))=\gamma \Dt. \ee
\item[]<4->

\vspace{.5cm}
Functional form of growth rate (extracts clock speed $\gamma$): \newline
\be 
g=\frac{\Dv}{\Dt}=\gamma.
\ee
\item[]<5->
\vspace{.5cm}
\underline{Behavioral protocol:} choose process with max $g$, outperform in the long run.
\ei
\vspace{.5cm}
}


\frame{{$v$-and-$x$ correspondence}

Role of $v(x)$: linearizes $x$ in time.
\vspace{1cm}

Can find $x$ as $x=v^{(-1)}(\gamma t)$.
}

\frame{{Stochastic growth}
Perturb deterministic growth consistently.
\vspace{.5cm}
\bi
\item[]<2->
Idea: add iid perturbation to linear process,
\vspace{.2cm}

$dv(x)=\gamma dt \hspace{.3cm} \rightarrow \hspace{.3cm} dv(x)=\gamma dt + \sigma dW.$
\vspace{.5cm}

\item[]<3->
Then find $x(t)$ as before as $x=v^{(-1)}(\gamma t + \sigma dW)$
\ei
}

\frame{
\bi
\item[]<1->
Measure speed of $x(t)$ like before: 
\vspace{.2cm}

$g=\frac{\Dv}{\Dt}=\gamma+\sigma \frac{W(\Dt)}{\Dt}$.
\vspace{.5cm}

\item[]<2->
Unlike before, $g$ is random, not constant in time.
\vspace{.5cm}

\item[]<3->


\begin{center}
\tikz\node[draw=red,line width=2pt,rounded corners=4pt,inner sep=6pt,align=center]{But $g$ is {\it \bf \color{red} ergodic}:\\ expected value  $\E{g}=\gbar$ time-average.};
\end{center}

\item[]<4->
\vspace{.5cm}
\underline{Behavioral protocol:} choose process with max $\gbar$, outperform in the long run.

\item[]<5->
\vspace{.5cm}
\underline{Note:} exact correspondence to deterministic case.
\ei


}

\frame{{Decision theory}
\bi
\setlength{\itemsep}{10pt}
\item[]<1->
1650s: expected-value theory -- maximize $\E{x(t+\Dt)}$.

\item[]<2->
1730s: doesn't work -- maximize $\E{u(x(t+\Dt))}$ instead.
\bi
\item[]<3->
Result: better describes observed behavior.

\item[]<4->
(Trivial: utility function $u(x)$ is extra degree of freedom.)
\ei
\item[]<5->
1940s: axiomatization -- max $\E{u(x(t+\Dt))}$ interpreted as consistent idiosyncratic preferences.
\ei
}

\frame{
    \bi
    \setlength{\itemsep}{10pt}
    \item[]<1->
Compare to our structure
    \item[]<2->
Maximand: $\gbar=\frac{\overline{\Dv}}{\Dt}.$
    \item[]<3->
Ergodic $\rightarrow$ time-average $\gbar=\E{g}$.
\vspace{.2cm}
\bi
\setlength{\itemsep}{8pt}
\item[]<4->
Maximize $\E{g}=\frac{\Dv}{\Dt}=\frac{1}{\Dt} \E{v(x(t+\Dt))}-v(x(t))$.
\item[]<5->
Multiply with constant $\Dt$, same behavior:\newline
$v(x(t+\Dt))-v(x(t))$.
\item[]<6->
add constant $v(x(t))$, same behavior:\newline
$\E{v(x(t+\Dt))}$.
\ei
\item[]<7->
\vspace{.2cm}
$\rightarrow$ expected-utility theory, except one letter
\begin{center}
$\E{\textcolor{red}{v}(x(t+\Dt))}=\E{\textcolor{red}{u}(x(t+\Dt))}$.
\end{center}
\ei
}

\ei
