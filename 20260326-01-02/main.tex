% ─── Slide 2: The misconception ───────────────────────────────────────────────
\begin{frame}{EE is often reduced to ``take the log'' — this misses the point}
\begin{columns}[T]
\column{0.46\textwidth}
\begin{alertblock}{\textcolor{EEred}{What people often think EE says}}
\vspace{0.3em}
Take the log of your wealth.\\
Maximise its expected value.\\[0.3em]
{\footnotesize\color{EEgrey} Equivalently: use log utility. The log is the whole story.}
\vspace{0.3em}
\end{alertblock}

\column{0.46\textwidth}
\begin{exampleblock}{\textcolor{EEgreen}{What EE actually says}}
\vspace{0.3em}
Find the \textbf{time-average growth rate} of your wealth under its \textbf{actual dynamics.}\\
Maximise that.\\[0.3em]
{\footnotesize\color{EEgrey} The log is one answer to ``what are the dynamics?'' — not the only one.}
\vspace{0.3em}
\end{exampleblock}
\end{columns}

% ─── Slide 3: Right clock ─────────────────────────────────────────────────────
\begin{frame}{A growth rate is only meaningful when measured on the right clock}

\begin{block}{EE axiom}
\small\centering
Maximize the \textbf{time-average growth rate}:
$g = \lim_{T\to\infty}\tfrac{1}{T}\left[\varphi\!\left(x(T)\right)-\varphi\!\left(x(0)\right)\right]$\\[0.15em]
{\footnotesize $\varphi$ is the \textbf{ergodicity transform} — the function that steadies the clock.}
\end{block}

\vspace{0.3em}
\begin{columns}[T]
\column{0.44\textwidth}
\textbf{Additive dynamics}
\[ x(t+1) = x(t) + \xi \]
\begin{itemize}\setlength\itemsep{3pt}
  \item Increment $\xi$ is independent of $x$
  \item Natural unit of change: $\Delta x$
  \item Growth rate: $g = \mathbb{E}[\xi]$
\end{itemize}

\[ x(t+1) = x(t)\cdot\eta \]
\begin{itemize}\setlength\itemsep{3pt}
  \item Return $\eta$ is independent of $x$
  \item Natural unit of change: $\Delta\ln x$
  \item Growth rate: $g = \mathbb{E}[\ln\eta]$
\end{itemize}

% ─── Slide 5a: Discounting — additive dynamics ────────────────────────────────
\begin{frame}{\$10 now or \$25 in a month? Additive world: timing is all that matters}

\begin{columns}[T]
\column{0.48\textwidth}
\vspace{0.4em}
A stranger offers you two certain payments:
\begin{itemize}\setlength\itemsep{3pt}
  \item Option 1: \$10 \textbf{now}
  \item Option 2: \$25 \textbf{in one month}
\end{itemize}

\vspace{0.6em}
Under \textbf{additive dynamics} ($x \to x + \xi$), the growth rate is the payment divided by the waiting time:
\[
  g_a^{(1)} = \frac{\$10}{1\,\text{month}}, \qquad g_a^{(2)} = \frac{\$25}{2\,\text{months}}
\]

\vspace{0.2em}
\begin{exampleblock}{}
\small $g_a^{(2)} = \$12.50/\text{month} > g_a^{(1)} = \$10/\text{month}$\\[0.3em]
\textbf{Always prefer \$25.} No dependence on wealth.
\end{exampleblock}

\vspace{0.4em}
{\small\color{EEgrey} The additive clock reads only the absolute increment — your current wealth is invisible to it.}

% ─── Slide 6: Bridge ──────────────────────────────────────────────────────────
\begin{frame}{The log clock works for multiplicative dynamics — but wealth can grow in other ways}

\begin{columns}[T]
\column{0.44\textwidth}
\vspace{0.4em}
So far: $x \to \eta\,x$ — each step \emph{multiplies} wealth.

\vspace{0.8em}
But some processes go further:\\
$x \to x^a$ — each step \emph{raises wealth to a power}.\\[0.3em]
{\small\color{EEgrey} Not $x$ scaled by a factor,\\but $x$ taken to an exponent.}

\vspace{0.8em}
On the log clock: $\ln(x^a) = a\ln x$ — the increment \emph{still depends on the current value}. The log clock cannot keep up.

\vspace{0.8em}
\begin{alertblock}{}
\small We need a faster clock.\\
What is it?
\end{alertblock}

\column{0.52\textwidth}

\end{columns}
\end{frame}

% ─── Slide 7: Power-law clock ────────────────────────────────────────────────
\begin{frame}{Power-law dynamics need a faster clock: $\varphi(x) = \ln\ln x$}

\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Finding the clock for $x \to x^a$:}

\vspace{0.4em}
Let $y = \ln x$.\quad $x \to x^a \;\Rightarrow\; y \to a\,y$ \quad — still multiplicative.\\[0.3em]
Let $z = \ln y = \ln\ln x$. \; $z \to \ln(a\,y) = \ln a + z$ \quad — \textcolor{EEgreen}{additive!}

\vspace{0.5em}
\[
  \boxed{\varphi(x) = \ln\ln x}
\]

\vspace{0.6em}
\textbf{Growth rate on the double-log clock:}
\[
  g = \ln a \quad \text{(constant — no dependence on } x \text{)}
\]

{\small\color{EEgrey} e.g.\ $x \to x^2$: $g = \ln 2$. \quad $x \to x^{1/3}$: $g = -\ln 3$.}

\column{0.46\textwidth}
\vspace{0.5em}
\begin{block}{Three dynamics — three clocks}
\vspace{0.3em}
\small
\begin{tabular}{lll}
\toprule
Dynamics & Clock $\varphi$ & $g$ \\
\midrule
$x \to x + \xi$ & $x$ & $\mathbb{E}[\xi]$ \\[4pt]
$x \to \eta\,x$ & $\ln x$ & $\mathbb{E}[\ln\eta]$ \\[4pt]
$x \to x^a$ & $\ln\ln x$ & $\mathbb{E}[\ln a]$ \\
\bottomrule
\end{tabular}
\vspace{0.2em}
\end{block}

\vspace{0.8em}
\end{columns}
\end{frame}

% ─── Slide 8: Absorbing boundaries ──────────────────────────────────────────
\begin{frame}{The clock stops at ruin — and that changes the optimal strategy}

\begin{columns}[T]
\column{0.46\textwidth}
\vspace{0.3em}
The time-average framework assumes the clock runs to infinity.

\vspace{0.6em}
But real processes have \textbf{absorbing states}: bankruptcy, ruin, extinction.

\vspace{0.6em}
Once absorbed, the clock stops permanently — no future growth can compensate.

\vspace{0.8em}
\begin{alertblock}{\textcolor{EEred}{Consequence}}
\small Maximising $g$ alone is not enough near a boundary.\\[0.3em]
The optimal strategy must weigh the growth rate \textbf{against} the probability that the clock stops before the long run arrives.
\end{alertblock}

% ─── Slide 9: Summary ────────────────────────────────────────────────────────
\begin{frame}{EE says: find the right clock for your dynamics, then maximise time-average growth}

\begin{columns}[T]
\column{0.52\textwidth}
\vspace{0.4em}
\textbf{Three take-home messages:}

\vspace{0.6em}
\begin{enumerate}\setlength\itemsep{10pt}
  \item \textbf{EE is not the log transform} \\ The log is the right clock \emph{for multiplicative dynamics}. Different dynamics demand different clocks.

  \item \textbf{Discount rates are not free parameters} \\ Your clock speed equals your discount rate. It is derived from the dynamics, not fitted to data.

  \item \textbf{The clock can stop} \\ Near absorbing boundaries, maximising $g$ alone is not enough — the probability of ruin changes what you should do.
\end{enumerate}
