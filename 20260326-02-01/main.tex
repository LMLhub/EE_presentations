\frametitle{Markets}
Economic agents can maximise their wealth over time by maximising the 
time-average growth rate of the wealth.
\vspace{1cm}
\bi
\item What does that mean for the portfolio choice?
  \bi \item Optimal leverage
  \ei
\vspace{1cm}
\item What does that mean for the way markets organise?
  \bi \item Leverage efficiency
  \ei
\ei
}

\frame{
\frametitle{From the cointoss to GBM}
  Make the time steps infinitesimally small and scale appropriately:
\be
  dx = x(\mu \, dt + \sigma \, dW\,)
\ee
\bi
  \item Drift: $\mu$
  \item Volatility: $\sigma$
\ei

\begin{figure}[H]
\fbox{
  \parbox{6cm}{
  Multiplicative dynamics: \\
  Ensemble average growth rate, $r$: $\mu$ \\
  Time average growth rate, $g$: $\mu - \frac{\sigma^2}{2}$
  }
}
\end{figure}
}

\frame{
\frametitle{Portfolio choice}
  Assume there are only two assets available
  \bi
    \item Risk-free asset:     
    \be 
    dx_r = x_r \, \mu_r \, dt 
    \ee
    \item Risky asset (stock):
    \be
        dx_s = x_s \, ( \mu_s \, dt + \sigma_s \, dW)
    \ee
        
  \ei
  \vspace{1cm}
  Portfolio problem: Set the relative share of wealth allocated to the stock 
  \bi
    \item Leverage, $l$
  \ei
  Wealth dynamic:
  \be
    dx = x(((1-l)\,\mu_r + l \,\mu_s )\, dt + l \, \sigma_s \, dW)
  \ee
  \bi
    \item Drift: $ \mu_r + l \, \mu_e $,
     where $\mu_e$ is the excess return, $\mu_s - \mu_r$
    \item Volatility: $l \, \sigma_s$
  \ei
}

\frame{
\frametitle{Optimal leverage}
  Wealth dynamic:
  \be
    dx = x((\mu_r + l \,\mu_e )\, dt + l \, \sigma_s \, dW\,.)
  \ee
  Time average growth rate: $g = \mu_r + l \, \mu_e - l^2  \,\frac{\sigma_s^2}{2}$ 
  
  Maximise $g$ with respect to $l$:
  \be
    l_{opt} = \frac{\mu_e}{\sigma_s^2}
  \ee
}
