\frame{
\frametitle{The Farmers Fable}
2 Farmers Ann and Bill each.
Each can have a good harvest ($g+r$) or a bad harvest ($g-r$) with equal probability.
They have the option to pool and share their harvests.


}

\frame{
\frametitle{Expected value analysis}
Without cooperation:
\begin{align*}
\mathbb{E} [\text{Ann's harvest}] &= \frac{1}{2}(g+r) + \frac{1}{2}(g-r) = g \\
\mathbb{E} [\text{Bill's harvest}] &= g
\end{align*}
}

\frame{
\frametitle{EE value analysis}
Without cooperation each farmer experiences the coin-toss dynamics which
gives  a time-averaged growth factor of:
\begin{align*}
G_\text{indiv} &= \sqrt{(g+r)(g-r)} = \sqrt{g^2 - r^2}
\end{align*}

With cooperation each farmer experiences  a time-averaged growth factor of:
\begin{align*}
G_\text{coop} &= \sqrt{g \sqrt{g^2 - r^2}}
\end{align*}

Can show that $G_\text{coop}  >  G_\text{indiv} $ for all allowed values of $g$ and $r$.
}


\frame{
\frametitle{An example where survival requires cooperation}
Choose $g=\sqrt{2}$, $r = \frac{2}{\sqrt{3}}$.

 $G_\text{indiv} <1$ but $G_\text{coop}  >1$

 Plot.

\bi
\item The \red{ensemble average} grows linearly: $\ave{x(t)} = \mu t$
\item Individual paths fluctuate around this mean
\item Higher $\sigma$ increases the spread of paths
\item The \red{time average} for a single path converges to $\mu t$ only as $t \to \infty$
\ei
\vf
This distinction between ensemble and time averages is central to ergodicity economics.
}

\frame{
\frametitle{Generalisation to $N$ members}
}


\frame{
\frametitle{Heterogeneity}
Suppose agent has it's own individual growth rate:
\begin{align*}
\bar{g}_i & = \mu_i -\frac{\sigma_i^2}{2}
\end{align*}
The growth rate of the cooperative is
\begin{align*}
\bar{g}^{(N)} &= \frac{1}{N} \sum_{i=1}^N \left( \mu_i - \frac{\sigma_i^2}{2 N}\right)
\end{align*}

}
