\frame{
\frametitle{Markets}
Economic agents can maximise their wealth over time by maximising the 
time-average growth rate of the wealth.
\vspace{1cm}
\bi
\item What does that mean for the portfolio choice?
  \bi \item Optimal leverage
  \ei
\vspace{1cm}
\item What does that mean for the way markets organise?
  \bi \item Leverage efficiency
  \ei
\ei
}

%----------------3-----------------
\frame{
\frametitle{Optimal leverage}
For a portfolio consisting of one risky asset ($s$) and one risk-free asset ($r$), the optimal share of wealth invested in the risky asset is:
\begin{align}
  \ell_{opt} = \frac{\mu_s-\mu_r}{\sigma_s^2}
\end{align}

\begin{figure}[H]
\fbox{
  \parbox{6cm}{
  \textbf{Leverage efficiency:} \\
  Markets self-organise around 
  \be 
    \ell_{opt} = 1
  \ee
  }
}
\end{figure}
}

%----------------4-----------------

\frame{
\frametitle{Leverage efficiency}
Why $\ell_{opt} = 1$?
\bi
  \item Efficiency: \n
  It is impossible for an individual investor to beat the market simply by using leverage.
  \item Stability: 
    \bi 
      \item $\ell_{opt} > 1$: Everyone wants to borrow cash to buy stocks, but noone wants to
      lend money.
      \item $\ell_{opt} < 0$: Everyone wants to borrow stocks and sell them to get cash, but
      noone wants to lend stocks.
    \ei
\ei
}

\frame{
\frametitle{Market simulation}
\textcolor{mplred}{$B$}'s rate setting rule:
\bi
  \item If \textcolor{mplred}{$B$}'s return was at least as good as \textcolor{mplblue}{$A$}'s during $dt$:
  \bi 
    \item Do nothing
  \ei 
  \item If \textcolor{mplred}{$B$}'s return was worse than \textcolor{mplblue}{$A$}'s during $dt$: 
  \bi 
    \item If \textcolor{mplblue}{$A$} borrowed from \textcolor{mplred}{$B$}: Increase $\mu_r$ with $d \mu $
    \item If \textcolor{mplred}{$B$} borrowed from \textcolor{mplblue}{$A$}: Decrease $\mu_r$ with $d \mu $
  \ei
\ei
\vf
\textcolor{mplblue}{$A$} and \textcolor{mplred}{$B$} start out with the same wealth.
}

\frame{
\frametitle{Equity Premium Puzzle}
Standard economic theory cannot explain the high premium on equity, but leverage efficiency can:
\bi
  \item $ ~3-10 \%$ in real life
  \item $ ~0.1-0.5 \% $ with standard theory (CCAPM)
  \item $ ~0.4-6.4 \% $ with optimal leverage ($s=$S\&P500 and $r =$ FED overnight interbank rate)
\ei


Premium: difference in time average growth rate ($\epsilon$)
\begin{align}
\epsilon &= \bar{g}_s - \bar{g}_r \\
        &= (\mu_s -\mu_r) - \frac{\sigma_s^2}{2} \\
        &= \frac{\mu_s -\mu_r}{2}
\end{align}
using leverage efficiency $\ell_{opt}=1 \implies \mu_s - \mu_r = \sigma_s^2$ 
}

%---------------- 9 -----------------
\frame{
  \frametitle{Summary}
  \bi
  \item For a risky and risk-free asset, the growth-rate optimal leverage is $\ell_{opt} = \frac{\mu_s - \mu_r}{\sigma_s^2}$
  \item Leverage efficient markets self-organise around $\ell_{opt} = 1$
  \item Leverage efficiency predicts a fixed relationship between drift, volatility and risk premium, supported by real life data
  \ei
}
